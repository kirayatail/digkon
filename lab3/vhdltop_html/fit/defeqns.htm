FTCPE_borrmotor: FTCPE port map (borrmotor,borrmotor_T,clk,'0','0');
     borrmotor_T <= ((NOT resin AND borrmotor)
      OR (puls AND borrmotor AND din(0) AND NOT din(2) AND NOT din(1))
      OR (resin AND puls AND NOT borrmotor AND NOT din(0) AND NOT din(2) AND
      NOT din(1)));
d1 <= NOT dav;
d2 <= NOT ((puls AND NOT din(0) AND NOT din(2) AND NOT din(1)));
d3 <= NOT ((puls AND din(0) AND NOT din(2) AND NOT din(1)));
FDCPE_levin: FDCPE port map (levin,q,clk,'0','0');
FDCPE_puls: FDCPE port map (puls,puls_D,clk,NOT resin,'0');
     puls_D <= ((puls AND x2/state_FSM_FFd1)
      OR (NOT x2/state_FSM_FFd1 AND NOT x2/state_FSM_FFd2 AND levin));
FDCPE_q: FDCPE port map (q,dav,clk,'0','0');
FTCPE_solenoid: FTCPE port map (solenoid,solenoid_T,clk,'0','0');
     solenoid_T <= ((NOT resin AND solenoid)
      OR (puls AND solenoid AND din(0) AND NOT din(2) AND din(1))
      OR (resin AND puls AND NOT solenoid AND NOT din(0) AND NOT din(2) AND
      din(1)));
FDCPE_x2/state_FSM_FFd1: FDCPE port map (x2/state_FSM_FFd1,x2/state_FSM_FFd1_D,clk,NOT resin,'0');
     x2/state_FSM_FFd1_D <= ((x2/state_FSM_FFd2)
      OR (x2/state_FSM_FFd1 AND levin));
FDCPE_x2/state_FSM_FFd2: FDCPE port map (x2/state_FSM_FFd2,x2/state_FSM_FFd2_D,clk,NOT resin,'0');
     x2/state_FSM_FFd2_D <= (NOT x2/state_FSM_FFd1 AND NOT x2/state_FSM_FFd2 AND levin);
